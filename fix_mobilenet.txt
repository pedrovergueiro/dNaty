# CORREÇÃO PARA MobileNetV3Large

class MobileNetV3Large(nn.Module):
    def __init__(self, num_classes=100):
        super().__init__()
        self.model = models.mobilenet_v3_large(weights=None)
        # Reconstruir o classifier completamente
        # MobileNetV3 original: Linear(960→1280) → Hardswish → Dropout → Linear(1280→1000)
        # Precisamos mudar apenas a última camada
        self.model.classifier = nn.Sequential(
            nn.Linear(960, 1280),
            nn.Hardswish(inplace=True),
            nn.Dropout(p=0.2, inplace=True),
            nn.Linear(1280, num_classes),  # MUDANÇA: 1000 → num_classes
        )

    def forward(self, x):
        return self.model(x)

    def count_params(self):
        return sum(p.numel() for p in self.parameters())
